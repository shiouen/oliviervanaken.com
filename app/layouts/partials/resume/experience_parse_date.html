{{/* Parse date YYYY/MM */}} {{- $split := split . "/" -}} {{- $yearStr := index $split 0 -}} {{ if findRE "^0" $yearStr }} {{ $yearStr = replaceRE "^0+" "" $yearStr }} {{ end }} {{- $year := int $yearStr -}} {{- $monthStr := index $split 1 -}} {{ if findRE "^0" $monthStr }} {{ $monthStr = replaceRE "^0+" "" $monthStr }} {{ end }} {{- $month := int $monthStr -}} {{- return dict "year" $year "month" $month -}}
